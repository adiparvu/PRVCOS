withMobileAuth() — Bearer Token Validation
// apps/web/src/lib/mobile/auth.ts

export type MobileHandler = (
  req: NextRequest,
  ctx: MobileContext
) => Promise<NextResponse>

export interface MobileContext {
  userId:    string
  companyId: string
  role:      string
  scopeLevel: string
  storeId?:  string
  regionId?: string
}

export function withMobileAuth(handler: MobileHandler) {
  return async (req: NextRequest) => {
    // 1. Extract Bearer token
    const authHeader = req.headers.get("authorization")
    if (!authHeader?.startsWith("Bearer "))
      return unauthorized()

    const token = authHeader.slice(7)

    // 2. Look up session in Redis (same store as web sessions)
    const session = await getSession(token)
    if (!session) return unauthorized()

    // 3. Extend TTL on activity
    await refreshSession(token, session)

    // 4. Invoke handler with typed context
    return handler(req, {
      userId:    session.userId,
      companyId: session.companyId,
      role:      session.role,
      scopeLevel: session.scopeLevel,
      storeId:   session.storeId,
      regionId:  session.regionId,
    })
  }
}

// apps/web/src/lib/mobile/command.ts

export async function assembleCommand(ctx: MobileContext): Promise<CommandData> {
  const scope = resolveScope(ctx)  // { companyIds, storeIds, regionId }

  // Run all queries in parallel for speed
  const [kpis, secondary, alerts, inbox] = await Promise.all([
    queryKPIs(scope, ctx.role),
    querySecondary(scope, ctx.userId),
    queryAlerts(scope),
    queryInbox(ctx.userId),
  ])

  return {
    user:         buildUser(ctx),
    kpis:         formatKPIs(kpis, ctx.role),
    secondary:    formatSecondary(secondary, ctx.role),
    aiBriefing:   buildAIBriefing(kpis, alerts, ctx.role),
    alerts:       formatAlerts(alerts),
    quickActions: QUICK_ACTIONS[ctx.role] ?? QUICK_ACTIONS.default,
    inbox:        formatInbox(inbox),
  }
}
