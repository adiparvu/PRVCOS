Section 05
Role-Aware Routing Architecture
// Role → Shell mapping (resolved at session hydration)
const ROLE_SHELLS = {
group_ceo: "executive", // ⌂ Command | ◎ Companies | ✦ Intelligence | ⟁ Finance | ⊞ Operations
ceo: "executive", // same tabs, SCOPE_COMPANY filtered
co_ceo: "executive", // same tabs, delegation scope
system_administrator: "sysadmin", // ⚙ System | 👤 Users | 🔐 Security | 📋 Audit | 🔗 Integrations
worker: "worker", // ⌂ Today | ✓ Tasks | 📅 Schedule | 📬 Inbox | 👤 Me
team_leader: "team_leader", // ⌂ Command | 👥 Team | ✓ Tasks | 📅 Schedule | 📬 Inbox
oms: "oms", // ⌂ Command | 👥 Area | ⊞ Operations | 📊 Reports | 📬 Inbox
operations_manager: "ops_manager",// ⌂ Command | ⊞ Operations | 👥 People | ⟁ Finance | 📊 Intelligence
department_head: "dept_head", // ⌂ Command | 👥 Team | ✓ Tasks | 📅 Schedule | 📬 Inbox
hr_payroll: "hr", // ⌂ Command | 👥 People | 💰 Payroll | 📄 Documents | 📬 Inbox
// ... project_worker, project_team_leader, project_oms,
// project_operations_manager, project_director,
// seller, store_manager, shop_director,
// app_support_specialist, data_analyst, qa_tester
}
// Session hydration on login
function resolveShell(session: Session): ShellConfig {
const shell = ROLE_SHELLS[session.role] ?? "worker"
return SHELL_CONFIGS[shell](session)
}
